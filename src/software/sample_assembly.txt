//Assembly code instruction set:

//beq operand => skip next instruction if operand = input
//bne operand => skip next instruction if operand != input
//blt operand => skip next instruction if operand < input
//bge operand => skip next instruction if operand >= input
//out data => output data
//jmp address => jump to address

//Sample Blinky program:
//(assuming input state stays at 0)

beq 0
jmp 0
bne 1
jmp 0
out 0
jmp 8

out 10
jmp 0

blt 3
jmp 8
bge 0
jmp 8
out 1
jmp 0
